    int binary_search(int sorted_list[], int low, int high, int element)
    {
        int middle;
        while(low <= high)
        {
            middle = (high-low)/2 +low;
            if(element < sorted_list[middle])
                high = middle - 1;
            else if( element > sorted_list[middle])
                low = middle + 1;
            else
                return middle;
    
        }
        return -1;
    }
